mysql之基础知识总结

连接方式(java)

driver:"jdbc:mysql://localhost:3306/db_name"
url:"com.mysql.jdbc.Driver"

表操作(create/drop/alter)

//创建
create table table_name(
    prop_name type，
    ...
);

//删除
drop table table_name;

//修改
//修改表名
alter table table_oldname rename table_newname;
//最后位置增加字段
alter table table_name add prop_name type;
//首位置增加字段
alter table table_name add prop_name type first;
//指定位置增加字段
alter table table_name add prop_name type after prop_name1;
//修改字段顺序
alter table table_name modify prop_name1 type first|after prop_name2;
//删除字段
alter table table_name drop prop_name;
//修改字段数据类型
alter table table_name modify prop_name type;
//修改字段名和数据类型
alter table table_name change prop_oldname prop_newname oldtype|newtype;

//查询
show/desc table_name
//通过系统表查看

//约束
    not null            //约束字段不能为空
    default                //设置字段默认值
    unique key(UK)        //设置字段值唯一
    primary key(PK)        //设置主键
    foreign key(FK)        //外键
    auto_increment        //字段值自动增加

//设置非空约束    
create table table_name(
    prop_name type not null，
    ...
);
//设置字段默认值    
create table table_name(
    prop_name type default value，
    ...
);
//设置唯一约束
create table table_name(
    prop_name type unique，
    ...
);
//设置主键约束(单主键)
create table table_name(
    prop_name type primary key，    //单字段主键
    ...
);
//设置主键约束    （多主键）
create table table_name(
    prop_name type,
    ...
    constraint primary    key(属性名,属性名...)
);
//设置字段值字段增加    
create table table_name(
    prop_name type auto_increment，
    ...
);
//设置外键约束
create table table_name(
    prop_name type,
    ...
    constraint 外键约束名 foreign key(prop_childname)    
        references 表名(prop_fathername)                    
);

索引操作

//创建表时创建普通索引
create table table_name(
    prop_name type,
    ...
    index|key index_name(prop_name [indexlength] [asc|desc])
);
//存在表时创建普通索引
create index index_name on table_name(prop_name [indexlength] [asc|desc]);
alter table table_name add index|key index_name(prop_name [indexlength] [asc|desc]);

//创建唯一索引（唯一性）
create table table_name(
    prop_name type,
    ...
    unique index|key index_name(prop_name [indexlength] [asc|desc])
);
//存在表时创建唯一索引（参上）

// 创建全文索引（快速查询数据量大的字符串类型）
create table table_name(
    prop_name type,
    ...
    fulltext index|key index_name(prop_name [indexlength] [asc|desc])
);
//创建多列索引
create table table_name(
    prop_name type,
    ...
    index|key index_name(prop_name [indexlength] [asc|desc], ...)
);
//删除索引
drop index index_name;
//修改索引（先删除再创建）

//查询索引
show create table table_name \G;
//校验
explain select * from t_dept where prop_name = 1;

视图操作

//创建视图
create view view_name as select...;
//删除视图
drop view view_name;
//修改视图
alter view view_name as select...;
//查询视图
show create view view_name;
desc view_name;
//通过系统表查看

触发器操作

//创建
create trigger trigger_name before|after trigger_EVENT on table_name for each row trigger_sql;
create trigger trigger_name before|after trigger_EVENT on table_name for each row 
    begin 
        trigger_sql,
        ...
    end
//删除
drop trigger trigger_name;
//查看
show triggers \G;
//通过系统表查看
use information_schema;
select * from triggers \G;

数据操作

//插入完整或部分记录
insert into table_name(col_name1, col_name2...)values(value1, value2...);
//插入多条记录
insert into table_name(col_name1, col_name2...)values(value1, value2...),
    values(value1, value2...),
    values(value1, value2...);
//插入查询记录
insert into table_name(col_name1, col_name2...)
    select (col_name1, col_name2...) from table_name where...
//更新数据
update table_name set col_name1 = value1, col_name2 = value2..., where...;
//删除数据
delete from table_name where...;
//清空表中数据（truncate相当于保留mysql表的结构，重新创建了这个表，效率高，不可恢复）
delete from table_name;
truncate table table_name;
//查询
//简单查询
select col_name1, col_name2... from table_name;
select * from table_name;
//去除重复
select distinct col_name1, col_name2... from table_name;
//设置显示格式
select concat(col_name1, 'str', col_name2) col_name2 from table;
//条件查询
select * from table_name where...;
select * from table_name where condition1 and condition2;
select * from table_name 
//范围查询
select * from table_name where col_name [not] between value1 and value2;
//集合查询
select * from table_name where col_name [not] in(value1, value2...);
//模块查询（%匹配多个字符|_匹配单个字符）
select * from table_name where col_name [not] like 'str%';
select * from table_name where col_name [not] like 'str_';
//排序(asc 升序|desc 降序)
select * from table_name where col_name1 order by [asc|desc][, col_name2 order by [asc|desc]]
//统计（count()|avg()|sum()|max()|min()）
select count(col_name) from table_name...;
...
//分组查询
select * from table_name where condition group by col_name;
//多表查询（内外连接）
//并(union all不去除重复)
select col_name1... from table_name1 union|union all select col_name2 from table_name2...
//笛卡尔积（记录数为多表相乘）
//内连接（笛卡尔积数据记录中，保留所有匹配记录，舍弃不匹配）
select col_name1... from table_name1 inner join table_name2 [inner join table_nameN] on|where condition; //貌似用where速度更快
select aliases_name1.col_name1, aliases_name2.col_name1 from table_name1 aliases_name1 inner join table_name2 [aliases_name2] [inner join table_nameN] on|where condition;
    //自然连接（笛卡尔积数据记录中根据表关系中相同名称的字段自动进行记录匹配，去掉重复字段）
    //等值连接（笛卡尔积数据记录中，选择所匹配字段值相等的记录，不会去掉重复）
    //不等连接（笛卡尔积数据记录中，选择所匹配字段值不相等的记录，不会去掉重复）
//外连接（笛卡尔积数据记录中，不仅保留所有匹配记录，还保留部分不匹配的数据记录）
select col_name... from table_name left|right|full [outer] join table_name2 where|on condition
